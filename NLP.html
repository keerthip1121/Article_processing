     "isn't", 'ma', 'mightn', "mightn't", 'mustn', "mustn't", 'needn', "needn't", 'shan', "shan't", 'shouldn', "shouldn't",
     'wasn', "wasn't", 'weren', "weren't", 'won', "won't", 'wouldn', "wouldn't"]

for word in sw:
    if(word not in nsw):
        new_sw.append(word)
for i in txt:
    if i not in new_sw:
        corpus.append(wc.lemmatize(i))
